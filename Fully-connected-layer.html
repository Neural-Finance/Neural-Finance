#define one layer
def dense_layer(inputs,name,units,w,mask,activation=None,use_bias=True,kernel_initializer=None):
        """Mimics tf.dense_layer but masks weights and uses presets as necessary."""
        # If there is a preset for this layer, use it.
        kernel_initializer = tf.constant_initializer(w)
        # Create the weights.
        weights = tf.get_variable(
            name=name + '_w',
            shape=[inputs.shape[1], units],
            initializer=kernel_initializer)
    
        # Mask the layer as necessary.
        mask_initializer = tf.constant_initializer(mask)
        mask = tf.get_variable(
              name=name + '_m',
              shape=[inputs.shape[1], units],
              initializer=mask_initializer,
              trainable=False)
        weights = tf.multiply(weights, mask)
        # Compute the output.
        output = tf.matmul(inputs, weights)
    
        # Add bias if applicable.
        if use_bias:
          bias = tf.get_variable(
              name=name + '_b', shape=[units], initializer=tf.zeros_initializer())
          output += bias
    
        # Activate.
        if activation:
          return activation(output),weights
        else:
          return output,weights

def initialize_uninitialized(self,sess):

    global_vars          = tf.global_variables()
    initialized   = sess.run([tf.is_variable_initialized(var) for var in global_vars])
    not_initialized_vars = [v for (v, f) in zip(global_vars, initialized) if not f]

    if len(not_initialized_vars):
        sess.run(tf.variables_initializer(not_initialized_vars))
